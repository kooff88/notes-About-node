## 内置模块

封装底层lib, index.js调用

./index.js
```js
const g = require('./lib');

g.addListener('newLesson', (res) => {
    if(res.price < 80) {
        console.log('买')
    }
    console.log('yeah!',res)

})
```

lib.js
```js
const EventEmitter = require('events').EventEmitter;

class Geektime extends EventEmitter {
    constructor(){
        super();
        setInterval(() => {
            this.emit('newLesson', { price: Math.random() * 100 })
        },3000)
    }


}

const g = new Geektime;

module.exports = g;

```
